def find(name, options)
  Rango::Bundling.strategies.each do |strategy|
    strategy = strategy.new(name, options)
    return strategy if strategy.match?
  end
  return nil
end

def activate_gem
  args = [@options[:gem]]
  version = @version || @options[:version]
  args.push(version) if version
  gem *args
end
